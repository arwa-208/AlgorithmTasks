Task1

START

  // Step 1: Read the list of numbers
  READ numbers_list

  // Step 2: Initialize sum and count
  sum ← 0
  count ← 0

  // Step 3: Go through each number in the list
  FOR each number IN numbers_list DO
      sum ← sum + number       // Add the number to the total sum
      count ← count + 1        // Increase the count of numbers
  END FOR

  // Step 4: Calculate the average
  average ← sum / count

  // Step 5: Return or display the average
  RETURN average

END

START

  READ number

  // Step 2: Check if the number is divisible by 2
  IF number MOD 2 = 0 THEN
      // Step 3: If divisible by 2, it is even
      RETURN "Even"
  ELSE
      // Step 4: Otherwise, it is odd
      RETURN "Odd"
  END IF

END
