Task1

START

  // Step 1: Read the list of numbers
  READ numbers_list

  // Step 2: Initialize sum and count
  sum ← 0
  count ← 0

  // Step 3: Go through each number in the list
  FOR each number IN numbers_list DO
      sum ← sum + number       // Add the number to the total sum
      count ← count + 1        // Increase the count of numbers
  END FOR

  // Step 4: Calculate the average
  average ← sum / count

  // Step 5: Return or display the average
  RETURN average

END

Task3
START

  // Step 1: Read the list of numbers
  READ numbers_list

  // Step 2: Assume the first number is the smallest
  smallest ← numbers_list[0]

  // Step 3: Go through each number in the list
  FOR each number IN numbers_list DO
      IF number < smallest THEN
          // Step 4: If the current number is smaller, update smallest
          smallest ← number
      END IF
  END FOR

  // Step 5: Return the smallest number
  RETURN smallest

END

Task4

START

  // Step 1: Read the list of numbers
  READ numbers_list

  // Step 2: Initialize counters
  positive_count ← 0
  negative_count ← 0

  // Step 3: Go through each number in the list
  FOR each number IN numbers_list DO
      IF number > 0 THEN
          positive_count ← positive_count + 1
      ELSE IF number < 0 THEN
          negative_count ← negative_count + 1
      END IF
      // Zero is ignored (neither positive nor negative)
  END FOR

  // Step 4: Return the counts
  RETURN positive_count, negative_count

END

  // Step 1: Read the list of numbers
  READ numbers_list

  // Step 2: Create an empty list to store unique numbers
  unique_list ← empty list

  // Step 3: Go through each number in the original list
  FOR each number IN numbers_list DO
      IF number is NOT in unique_list THEN
          // Step 4: If number is not already in unique_list, add it
          ADD number TO unique_list
      END IF
  END FOR

  // Step 5: Return the list with unique values
  RETURN unique_list

END

Task6
START

  // Step 1: Read the number N
  READ N

  // Step 2: Create an empty list to store prime numbers
  prime_list ← empty list

  // Step 3: Check each number from 2 to N
  FOR number FROM 2 TO N DO
      is_prime ← true  // Assume the number is prime

      // Step 4: Check if number is divisible by any number from 2 to number-1
      FOR i FROM 2 TO number-1 DO
          IF number MOD i = 0 THEN
              is_prime ← false   // Number is not prime
              BREAK              // No need to check further
          END IF
      END FOR

      // Step 5: If number is prime, add it to prime_list
      IF is_prime = true THEN
          ADD number TO prime_list
      END IF
  END FOR

  // Step 6: Return the list of prime numbers
  RETURN prime_list

END

task7
START

  // Step 1: Read the two sorted lists
  READ list1
  READ list2

  // Step 2: Create an empty list for the result
  merged_list ← empty list

  // Step 3: Set pointers for both lists
  i ← 0
  j ← 0

  // Step 4: Compare elements from both lists
  WHILE i < length(list1) AND j < length(list2) DO
      IF list1[i] ≤ list2[j] THEN
          ADD list1[i] TO merged_list
          i ← i + 1
      ELSE
          ADD list2[j] TO merged_list
          j ← j + 1
      END IF
  END WHILE

  // Step 5: Add remaining elements from list1 (if any)
  WHILE i < length(list1) DO
      ADD list1[i] TO merged_list
      i ← i + 1
  END WHILE

  // Step 6: Add remaining elements from list2 (if any)
  WHILE j < length(list2) DO
      ADD list2[j] TO merged_list
      j ← j + 1
  END WHILE

  // Step 7: Return the merged sorted list
  RETURN merged_list

END

Task8

START

  // Step 1: Read the list of numbers
  READ numbers_list

  // Step 2: Initialize the largest and second largest values
  largest ← -∞
  second_largest ← -∞

  // Step 3: Go through each number in the list
  FOR each number IN numbers_list DO

      IF number > largest THEN
          second_largest ← largest
          largest ← number

      ELSE IF number < largest AND number > second_largest THEN
          second_largest ← number
      END IF

  END FOR

  // Step 4: Return the second largest number
  RETURN second_largest

END

  text ← convert text to lowercase

  // Step 3: Remove all spaces from the string
  text ← remove spaces from text

  reversed_text ← reverse text

  // Step 5: Compare the original and reversed strings
  IF text = reversed_text THEN
      RETURN "Palindrome"
  ELSE
      RETURN "Not a Palindrome"
  END IF

END

A palindrome is a word, phrase, number, or sequence that reads the same forward and backward.

Task10

START

  // Step 1: Read the number N
  READ N

  // Step 2: Create an empty list to store Fibonacci numbers
  fibonacci_list ← empty list

  // Step 3: Handle the first Fibonacci number
  IF N ≥ 1 THEN
      ADD 0 TO fibonacci_list
  END IF

  // Step 4: Handle the second Fibonacci number
  IF N ≥ 2 THEN
      ADD 1 TO fibonacci_list
  END IF

  // Step 5: Generate remaining Fibonacci numbers
  FOR i FROM 3 TO N DO
      next_number ← fibonacci_list[i-2] + fibonacci_list[i-3]
      ADD next_number TO fibonacci_list
  END FOR

  // Step 6: Return the Fibonacci sequence
  RETURN fibonacci_list

END
