Task1

START

  // Step 1: Read the list of numbers
  READ numbers_list

  // Step 2: Initialize sum and count
  sum ← 0
  count ← 0

  // Step 3: Go through each number in the list
  FOR each number IN numbers_list DO
      sum ← sum + number       // Add the number to the total sum
      count ← count + 1        // Increase the count of numbers
  END FOR

  // Step 4: Calculate the average
  average ← sum / count

  // Step 5: Return or display the average
  RETURN average

END

Task2

START

  // Step 1: Read the number
  READ number

  // Step 2: Check if the number is divisible by 2
  IF number MOD 2 = 0 THEN
      // Step 3: If divisible by 2, it is even
      RETURN "Even"
  ELSE
      // Step 4: Otherwise, it is odd
      RETURN "Odd"
  END IF

END

Task3
START

  // Step 1: Read the list of numbers
  READ numbers_list

  // Step 2: Assume the first number is the smallest
  smallest ← numbers_list[0]

  // Step 3: Go through each number in the list
  FOR each number IN numbers_list DO
      IF number < smallest THEN
          // Step 4: If the current number is smaller, update smallest
          smallest ← number
      END IF
  END FOR

  // Step 5: Return the smallest number
  RETURN smallest

END

Task4

START

  // Step 1: Read the list of numbers
  READ numbers_list

  // Step 2: Initialize counters
  positive_count ← 0
  negative_count ← 0

  // Step 3: Go through each number in the list
  FOR each number IN numbers_list DO
      IF number > 0 THEN
          positive_count ← positive_count + 1
      ELSE IF number < 0 THEN
          negative_count ← negative_count + 1
      END IF
      // Zero is ignored (neither positive nor negative)
  END FOR

  // Step 4: Return the counts
  RETURN positive_count, negative_count

END

Task5

START

  // Step 1: Read the list of numbers
  READ numbers_list

  // Step 2: Create an empty list to store unique numbers
  unique_list ← empty list

  // Step 3: Go through each number in the original list
  FOR each number IN numbers_list DO
      IF number is NOT in unique_list THEN
          // Step 4: If number is not already in unique_list, add it
          ADD number TO unique_list
      END IF
  END FOR

  // Step 5: Return the list with unique values
  RETURN unique_list

END
